We implement two different ways to calculate the variance and average, one is based on the list structure and the other using the numpy structure.

The skewness and Kurtosis functions are made by using the numpy structure.

def variance(x):
    ave=average(x)
    suma=0.
    for i in range(len(x)):
        suma+=(x[i]-ave)**2
    suma/=len(x)
    return suma

def variance2(x):

In [15]:
def average(x):
    suma=0.
    for i in range(len(x)):
        suma+=x[i]
    suma=suma/len(x)
    return suma
In [16]:
def average2(x):
    suma=np.sum(x)/len(x)
    return suma
In [17]:
def skewness(x):
    ave=average2(x)
    suma=np.sum((x-ave)**3.)/len(x)
    return suma
In [18]:
def Kurtosis(x):
    ave=average2(x)
    suma=np.sum((x-ave)**4.)/len(x)
    return suma
