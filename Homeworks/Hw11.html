This problem can be sovlved easily using the function np.linalg.eigvals(), such that

In [12]:

n=100
eigen_vals=[]
for i in range(n):
    matrix=np.random.normal(0,1,(m,m))
    matrix=matrix+matrix.T
    eigen_vals=np.append(eigen_vals,np.linalg.eigvals(matrix))
